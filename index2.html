10. map() and filter()

const nums = [1, 2, 3, 4];
const doubled = nums.map(n => n * 2); // [2, 4, 6, 8]
const evens = nums.filter(n => n % 2 === 0); // [2, 4]

11. Count 0's in Array

const arr = [1, 0, 1, 0, 0];
const zeros = arr.filter(n => n === 0).length;
console.log("0s:", zeros);

12. Count Odd & Even Numbers

const nums = [1, 2, 3, 4, 5, 6];
let odd = 0, even = 0;
nums.forEach(n => n % 2 === 0 ? even++ : odd++);
console.log("Even:", even, "Odd:", odd);

13. Count Vowels in String

const str = "hello world";
const vowels = str.match(/[aeiou]/gi);
console.log("Vowels:", vowels ? vowels.length : 0);
